async def main(args: Args) -> Output:
    params = args.params
    input_text = params['input']

    # 使用正则表达式按"场景"切分输入
    scenes = re.split(r'场景\d+：', input_text)

    # 去除空字符串
    scenes = [scene.strip() for scene in scenes if scene.strip()]

    # 构建输出对象
    ret: Output = {
        "scenes": scenes,
        "scene_count": len(scenes),
    }
    return ret
